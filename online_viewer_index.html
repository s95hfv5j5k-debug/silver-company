      /* ======================================================
      IMAGE PERSISTENCE — QUOTA-SAFE (avoids QuotaExceededError)
      - Compresses images before saving to localStorage
      - Limits per-item size and total budget
      - Falls back to sessionStorage (per-tab) or in-memory cache
      - Provides import/export helpers (triggered via console)

      const IMG_MAX_W = 1600; // px
      const IMG_MAX_H = 1000; // px
      const JPEG_QUALITY = 0.72; // initial quality
      const LS_ITEM_LIMIT = 280 * 1024; // ~280 KB per item
      const LS_TOTAL_LIMIT = 2 * 1024 * 1024; // ~2 MB soft budget
      const imageCache = new Map(); // in-memory fallback
      function estBytes(str){ return new Blob([str]).size; }
      function lsUsed(){
      let sum = 0; for(let i=0;i<localStorage.length;i++){ const k=localStorage.key(i); const v=localStorage.getItem(k)||''; sum += k.length + v.length; }
      return sum;

      function drawCompressed(dataUrl, maxW=IMG_MAX_W, maxH=IMG_MAX_H, q=JPEG_QUALITY){
      return new Promise((resolve, reject)=>{
      const img = new Image();
      img.onload = ()=>{
      const ratio = Math.min(1, maxW/img.width, maxH/img.height);
      const w = Math.max(1, Math.round(img.width*ratio));
      const h = Math.max(1, Math.round(img.height*ratio));
      const c = document.createElement('canvas'); c.width=w; c.height=h;
      const ctx = c.getContext('2d');
      ctx.drawImage(img,0,0,w,h);
      try{ const out = c.toDataURL('image/jpeg', q); resolve(out); }
      catch(err){ reject(err); }
      };
      img.onerror = reject;
      img.src = dataUrl;
      });
      }
      async function safeStoreImage(key, dataUrl){
      let out = await drawCompressed(dataUrl);
      if(estBytes(out) > LS_ITEM_LIMIT){ out = await drawCompressed(dataUrl, 1280, 900, 0.6); }
      try{
      if(lsUsed() + estBytes(out) > LS_TOTAL_LIMIT){
      const keys = Object.keys(localStorage).filter(k=>/^sc-(hero|g\d+)$/.test(k)).sort();
      while(keys.length && (lsUsed() + estBytes(out) > LS_TOTAL_LIMIT)){
      const k = keys.shift(); localStorage.removeItem(k);
      }
      }
      localStorage.setItem(key, out);
      }catch(e){
      try{ sessionStorage.setItem(key+':session', out); }catch(_){ }
      imageCache.set(key, dataUrl);
      console.warn('Quota hit. Using session/memory for', key);
      }
